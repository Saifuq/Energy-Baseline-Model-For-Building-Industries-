Load and merge in Python
import pandas as pd
train = pd.read_csv('train.csv')
weather = pd.read_csv('weather_train.csv')
bldg = pd.read_csv('building_metadata.csv')

df = train.merge(bldg, on='building_id', how='left')
df['timestamp'] = pd.to_datetime(df['timestamp'])
df = df.merge(weather, on=['site_id','timestamp'], how='left')
print(df.shape) # (20216100, ~15 columns)
print(df.head())
